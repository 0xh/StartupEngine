Note that for using any of these features you will also require the Bootstrap JS loaded in your project, following the Bootstrap requirements of each component, for example, the Popover component needs Popper.js or bootstrap.bundle.js to be loaded.

Custom popover options

You can use any Bootstrap Popover options (except: trigger, content and html) using the "popover" property in the colorpicker options.

Code
<div id="cp1" class="input-group">
  <input type="text" class="form-control input-lg" value="hsl(103, 70%, 64%)"/>
  <span class="input-group-append">
    <span class="input-group-text colorpicker-input-addon"><i></i></span>
  </span>
</div>
<script>
  $(function () {
    $('#cp1').colorpicker({
      popover: {
        title: 'Adjust the color',
        placement: 'top'
      }
    });
  });
</script>

Inside a modal

Modal title
Code
<!-- Button trigger modal -->
<button type="button" class="btn btn-primary" data-toggle="modal" data-target="#exampleModal">
  Launch demo modal
</button>

<!-- Modal -->
<div class="modal fade" id="exampleModal" tabindex="-1" role="dialog"
     aria-labelledby="exampleModalLabel" aria-hidden="true">
  <div class="modal-dialog" role="document">
    <div class="modal-content">
      <div class="modal-header">
        <h5 class="modal-title" id="exampleModalLabel">Modal title</h5>
        <button type="button" class="close" data-dismiss="modal" aria-label="Close">
          <span aria-hidden="true">&times;</span>
        </button>
      </div>
      <div class="modal-body">
        <input id="cp2" type="text" class="form-control input-lg" value="rgb(130, 44, 29)"/>
      </div>
      <div class="modal-footer">
        <button type="button" class="btn btn-secondary" data-dismiss="modal">Close</button>
        <button type="button" class="btn btn-primary">Save changes</button>
      </div>
    </div>
  </div>
</div>
<script>
  $(function () {
    $('#cp2').colorpicker();
  });
</script>
